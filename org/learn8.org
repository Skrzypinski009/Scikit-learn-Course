* Cross validation Framework

** Set up
#+begin_src python
from sklearn.datasets import fetch_california_housing

housing = fetch_california_housing(as_frame=True)
X, y = housing.data, housing.target
print(housing.DESCR)
#+end_src

#+begin_src python
y *= 100
print(y)
#+end_src

** DecisionTreeRegressor
#+begin_src python
from sklearn.tree import DecisionTreeRegressor

regressor = DecisionTreeRegressor(random_state=0)
print(regressor.fit(X, y))
#+end_src

** Calculating mean absolute error
#+begin_src python
from sklearn.metrics import mean_absolute_error

y_predicted = regressor.predict(X)
score = mean_absolute_error(y, y_predicted)
print(score)
#+end_src

** Splitting the data
#+begin_src python
from sklearn.model_selection import train_test_split

X_train, X_test, y_train, y_test = train_test_split(
    X, y, random_state=0
)
#+end_src

** Calculating mean absolute error on test data
#+begin_src python
regressor = DecisionTreeRegressor(random_state=0)
regressor.fit(X_train, y_train)
prediction = regressor.predict(X_test)
score = mean_absolute_error(y_test, prediction)
print(score)
#+end_src

** ShuffleSplit with cross validation
#+begin_src python
from sklearn.model_selection import cross_validate, ShuffleSplit
import pandas as pd

cv = ShuffleSplit(n_splits=40, test_size=0.3, random_state=42)
cv_results = cross_validate(regressor, X, y, cv=cv, scoring="neg_mean_absolute_error")
cv_results = pd.DataFrame(cv_results)
cv_results["test_error"] = -cv_results["test_score"]
print(cv_results.head())
#+end_src

** Error distribution
#+begin_src python :results graphics file :file ../img/learn8_plot1.png
import matplotlib.pyplot as plt

plt.clf()
cv_results["test_error"].plot.hist(bins=10, edgecolor="black")
plt.xlabel("Mean absolute error (k$)")
plt.title("Test error distribution")
plt.gcf()
#+end_src

#+begin_src python

)
