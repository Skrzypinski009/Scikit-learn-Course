import seaborn as sns

n_samples = 5000
columns = ["age", "education-num", "hours-per-week"]
plt.clf()
sns.pairplot(
    data=X[:n_samples],
    vars=columns,
    hue=target_column,
    plot_kws={"alpha": 0.2},
    height=3,
    diag_kind="hist",
    diag_kws={"bins": 30},
)
plt.gcf()
#+end_src

** Scatterplot
#+begin_src python :results graphics file :file ../img/learn1_plot3.png
plt.clf()
plt.figure(figsize=(5, 4))
sns.scatterplot(
    x="age",
    y="hours-per-week",
    data=X[:n_samples],
    hue=target_column,
    alpha=0.5,
)

age_limit = 27
plt.axvline(x=age_limit, ymin=0, ymax=1, color="black", linestyle="--")

hours_per_week_limit = 40
plt.axhline(y=hours_per_week_limit, xmin=0.18, xmax=1, color="black", linestyle="--")

plt.gcf()
