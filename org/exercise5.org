import pandas as pd

df = pd.read_csv("../datasets/adult_census.csv")
target_name = "class"
X, y = df.drop(columns=[target_name, "education-num"]), df[target_name]
#+end_src

#+begin_src python
from sklearn.compose import make_column_selector

categorical_selector = make_column_selector(dtype_include=object)
numerical_selector = make_column_selector(dtype_exclude=object)

categorical_columns = categorical_selector(X)
numerical_columns = numerical_selector(X)
#+end_src

#+begin_src python
import time
from sklearn.model_selection import cross_validate
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import OrdinalEncoder
from sklearn.compose import make_column_transformer
from sklearn.ensemble import HistGradientBoostingClassifier
#+end_src

#+begin_src python
categorical_preprocessor = OrdinalEncoder(
    handle_unknown="use_encoded_value",
    unknown_value=-1,
    encoded_missing_value=-1,
)

preprocessor = make_column_transformer(
    (categorical_preprocessor, categorical_columns),
    remainder="passthrough",
)

model = make_pipeline(
    preprocessor,
    HistGradientBoostingClassifier(),
)

start = time.time()
cv_results = cross_validate(model, X, y, cv=5)
elapsed_time = time.time() - start

scores = cv_results["test_score"]

print(
    f"Mean cross validation accuracy is:",
    f"{scores.mean():.3f} +- {scores.std():.3f}",
    f"with a fitting time of {elapsed_time:.3f} seconds"
)
#+end_src
