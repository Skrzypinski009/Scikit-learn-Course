** Set up
#+begin_src python :results none
import pandas as pd

blood_transfusion = pd.read_csv("../datasets/blood_transfusion.csv")

target_name = "Class"
X = blood_transfusion.drop(columns=[target_name])
y = blood_transfusion[target_name]
#+end_src

** Creating model
*StandardScaler* + *SVC* (Support Vector Classification)
#+begin_src python :results none
from sklearn.preprocessing import StandardScaler
from sklearn.svm import SVC
from sklearn.pipeline import make_pipeline

model = make_pipeline(
    StandardScaler(),
    SVC(kernel="rbf"),
)
#+end_src

** ShuffleSplit and cross validation
#+begin_src python :results none
from sklearn.model_selection import ShuffleSplit, cross_validate

cv = ShuffleSplit(random_state=0)

cv_results = cross_validate(
    model,
    X,
    y,
    cv=cv,
)
#+end_src

** ValidationCurveDisplay
Visualization of change of the accuracy with change of the *SVC gamma* parameter.
#+begin_src python :results graphics file :file ../img/exercise6_plot1.png
from sklearn.model_selection import ValidationCurveDisplay
import matplotlib.pyplot as plt
import numpy as np

plt.clf()

p_range = np.linspace(-3, 5, num=30)
display = ValidationCurveDisplay.from_estimator(
    model,
    X,
    y,
    cv=cv,
    param_name="svc__gamma",
    param_range=p_range,
    n_jobs=2,
    scoring="accuracy",
    std_display_style="errorbar",
    errorbar_kw={"alpha": 0.7},
)
plt.gcf()
#+end_src

    model,

    cv=cv,
    score_type="both",
    scoring="accuracy",
    score_name="Mean Absolute Error (k$)",
    std_display_style="errorbar",

)
