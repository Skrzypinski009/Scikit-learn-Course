* Using numerical and categorical variables together

** Set up
#+begin_src python
import pandas as pd

df = pd.read_csv("../datasets/adult_census.csv")
target_name = "class"
X, y = df.drop(columns=[target_name]), df[target_name]
#+end_src

** Selection based on type
#+begin_src python
from sklearn.compose import make_column_selector as selector

numerical_column_selector = selector(dtype_exclude=object)
categorical_column_selector = selector(dtype_include=object)

numerical_columns = numerical_column_selector(X)
categorical_columns = categorical_column_selector(X)
#+end_src

** Dispatch columns to a specific processor
#+begin_src python
from sklearn.preprocessing import OneHotEncoder, StandardScaler

categorical_preprocessor = OneHotEncoder(handle_unknown="ignore")
numerical_preprocessor = StandardScaler()
#+end_src

** Creating transformer
#+begin_src python
from sklearn.compose import make_column_transformer

preprocessor = make_column_transformer(
    (categorical_preprocessor, categorical_columns),
    (numerical_preprocessor, numerical_columns),
)
#+end_src

** Creating pipeline model
#+begin_src python
from sklearn.linear_model import LogisticRegression
from sklearn.pipeline import make_pipeline

model = make_pipeline(
    preprocessor,
    LogisticRegression(max_iter=500),
)
#+end_src

** Splitting the data
#+begin_src python
from sklearn.model_selection import train_test_split

X_train, X_test, y_train, y_test = train_test_split(
    X, y, random_state=42,
)
#+end_src

** Prediction and accuracy
#+begin_src python
model.fit(X_train, y_train)
prediction = model.predict(X_test[:5])
print(f"Prediction result: {(prediction == y_test[:5]).sum()} / 5")
#+end_src

#+begin_src python
accuracy = model.score(X_test, y_test)
print(f"Model accuracy: {accuracy:.3f}")
#+end_src

** Cross validation
#+begin_src python
from sklearn.model_selection import cross_validate

cv_results = cross_validate(model, X, y, cv=5)

for key, val in cv_results.items():
    print(f"| {key:<15} | {val} |")
#+end_src

#+begin_src python
print(f"Generalization performance")
print("+-------------+-------+")
print(f"| Test score: | {cv_results['test_score'].mean():.3f} |")
print(f"| Fit time:   | {cv_results['fit_time'].mean():.3f} |")
print(f"| Score time: | {cv_results['score_time'].mean():.3f} |")
print("+-------------+-------+")
