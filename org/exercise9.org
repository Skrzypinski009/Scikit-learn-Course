import pandas as pd

penguins = pd.read_csv("../datasets/penguins_regression.csv")
feature_name = "Flipper Length (mm)"
target_name = "Body Mass (g)"
X, y = penguins[[feature_name]], penguins[[target_name]]
y
#+end_src

** Linear model prediction function
#+begin_src python
def linear_model_flipper_mass(
    flipper_length, weight_flipper_length, intercept_body_mass
):
    """Linear model of the form y = a * x + b"""
    body_mass = weight_flipper_length * flipper_length + intercept_body_mass
    return body_mass
#+end_src

#+begin_src python
import numpy as np

flipper_length_range = np.linspace(X.min(), X.max(), num=300)
#+end_src

** Visualization different values of coefficients and intercept
#+begin_src python :results graphics file :file ../img/exercise9_plot1.png
import seaborn as sns
import matplotlib.pyplot as plt

weights = [45, 44, 47]
intercepts = [-5000, -4900, -5100]

plt.clf()
ax = sns.scatterplot(
    data=penguins, x=feature_name, y=target_name, color="black", alpha=0.5
)

for weight, intercept in zip(weights, intercepts):
    predicted_body_mass = linear_model_flipper_mass(
        flipper_length_range, weight, intercept
    )
    ax.plot(flipper_length_range, predicted_body_mass)

plt.gcf()

** Calculating error
#+begin_src python
def goodness_fit_measure(true_values, predictions):
    errors = predictions[feature_name] - true_values[target_name]
    return np.mean(np.abs(errors))

#+end_src
