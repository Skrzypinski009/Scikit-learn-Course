** LinearRegression with PolynomialFeatures
#+begin_src python
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import PolynomialFeatures

model = make_pipeline(
    PolynomialFeatures(degree=2, interaction_only=True, include_bias=False),
    LinearRegression(),
)
model.set_output(transform="pandas")
#+end_src

*** Cross validation

from sklearn.kernel_approximation import Nystroem

model = make_pipeline(
    Nystroem(kernel="poly", degree=2, random_state=0),
    LinearRegression(),
)
#+end_src

*** Visualization of inpact of n_components on MAE
#+begin_src python :results graphics file :file ../img/exercise10_plot1.png
from sklearn.model_selection import ValidationCurveDisplay
import matplotlib.pyplot as plt
import numpy as np

param_range = np.array([5, 10, 50, 100])
plt.clf()
display = ValidationCurveDisplay.from_estimator(
    model,
    data,
    target,
    param_name="nystroem__n_components",
    param_range=param_range,
    negate_score=True,
    std_display_style="errorbar",
    errorbar_kw={"alpha": 0.5},
    scoring="neg_mean_absolute_error",
)
plt.gcf()
