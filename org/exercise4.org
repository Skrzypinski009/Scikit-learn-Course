** Comparing OrdinalEncoder and OneHotEncoder
#+begin_src python :results none
from sklearn.pipeline import make_pipeline
from sklearn.preprocessing import OrdinalEncoder
from sklearn.linear_model import LogisticRegression
#+end_src

*** OrdinalEncoder
#+begin_src python :results none
model = make_pipeline(

    OrdinalEncoder(
        handle_unknown="use_encoded_value",
        unknown_value=-1,
        encoded_missing_value=-1,
    ).set_output(transform="pandas"),

    LogisticRegression(max_iter=500),

)
#+end_src

#+begin_src python :results output
from sklearn.model_selection import cross_validate

cv_results = cross_validate(model, X_categorical, y)

for key, val in cv_results.items():
    print(f"| {key:<10} | {val} |")
#+end_src

*** OneHotEncoder
#+begin_src python :results output :time
from sklearn.preprocessing import OneHotEncoder

model_ohe = make_pipeline(
    OneHotEncoder(handle_unknown="ignore"),
    LogisticRegression(max_iter=500)
)

cv_results_ohe = cross_validate(model_ohe, X_categorical, y)

for key, val in cv_results_ohe.items():
    print(f"| {key:<10} | {val} |")
#+end_src

#+begin_src python :results output
scores = cv_results['test_score']
fit_time = cv_results['fit_time']
score_time = cv_results['score_time']
print("---OrdinalEncoder")
print(f"accuracy: {scores.mean():.3f} +- {scores.std():.3f}")
print(f"fit_time: {fit_time.mean():.3f}")
print(f"score_time: {score_time.mean():.3f}")


scores_ohe = cv_results_ohe['test_score']
fit_time_ohe = cv_results_ohe['fit_time']
score_time_ohe = cv_results_ohe['score_time']
print("\n---OneHotEncoder")
print(f"accuracy: {scores_ohe.mean():.3f} +- {scores_ohe.std():.3f}")
print(f"fit_time: {fit_time_ohe.mean():.3f}")
print(f"score_time: {score_time_ohe.mean():.3f}")
#+end_src

Beside the score time, the *OneHotEncoder* has better results than *OrdinalEncoder*.
