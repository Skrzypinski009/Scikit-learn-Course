* Set and get hyperparameters in scikit-learn

** Set up
#+begin_src python
import pandas as pd

adult_census = pd.read_csv("../datasets/adult_census.csv")

target_name = "class"
numerical_columns = ["age", "capital-gain", "capital-loss", "hours-per-week"]

y = adult_census[target_name]
X = adult_census[numerical_columns]
X.head()
#+end_src

** Creating pipeline
#+begin_src python
from sklearn.pipeline import Pipeline
from sklearn.preprocessing import StandardScaler
from sklearn.linear_model import LogisticRegression

model = Pipeline(
    steps=[
        ("preprocessor", StandardScaler()),
        ("classifier", LogisticRegression()),
    ]
)
#+end_src

** Cross validation
#+begin_src python
from sklearn.model_selection import cross_validate

cv_results = cross_validate(model, X, y)
scores = cv_results['test_score']
f"Accuracy: {scores.mean():.3f} +- {scores.std():.3f}"
#+end_src

** Cross validation after changing a parameter of the model
#+begin_src python
model.set_params(classifier__C=1e-3)
cv_results = cross_validate(model, X, y)
scores = cv_results['test_score']
f"Accuracy: {scores.mean():.3f} +- {scores.std():.3f}"
#+end_src

** Getting all parameters of the model
#+begin_src python :results output
for parameter in model.get_params():
    print(parameter)
#+end_src

#+begin_src python
model.get_params()["classifier__C"]
#+end_src

** Testing different values of the C parameter in cross validation
#+begin_src python :results output
for C in [1e-3, 1e-2, 1e-1, 1, 10]:
    model.set_params(classifier__C=C)
    cv_results = cross_validate(model, X, y)
    scores = cv_results["test_score"]
    print(f"Accuracy: {scores.mean():.3f} +- {scores.std():.3f}")
#+end_src
